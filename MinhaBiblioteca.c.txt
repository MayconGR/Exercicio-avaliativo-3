#include <stdio.h>
#include "MinhaBiblioteca.h"

void numeros(int vetor[], int *maximo, int *minimo, int numElementos){
//Se o numero de elementos escolhidos para o vetor for 1:
  if(numElementos == 1){
    *maximo = vetor[0];
    *minimo = *maximo;
  }

//Lógica base para os dois primeiros elementos do vetor:
  if(vetor[0] > vetor[1]){
    *maximo = vetor[0];
    *minimo = vetor[1];
    
  }else{
    *maximo = vetor[1];
    *minimo = vetor[0];
  }

//Estrutura de repetição para as variaveis armazenarem o maior e o menor valor de um vetor:
  for(int i = 2; i<numElementos; i++){
    if(vetor[i] > *maximo){
      *maximo = vetor[i];
    }
    if(vetor[i] < *minimo){
      *minimo = vetor[i];
    }
  }
}
